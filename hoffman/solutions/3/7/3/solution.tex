\begin{align}
\vec{T}^tf(\vec{A})=f(\vec{T}(\vec{A}))\\
= f(\vec{AB} - \vec{BA})\\
=trace(\vec{AB} - \vec{BA})\label{eq:1}
\end{align}
using $trace(\vec{AB})=trace(\vec{BA})$ in \eqref{eq:1},
\begin{align}
\vec{T}^tf(\vec{A}) = trace(\vec{AB}) - trace(\vec{BA})= 0
\end{align}
Hence $\vec{T}^tf = 0$
