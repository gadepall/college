       \parbox{10cm}{\begin{align}

         \vec{c}=\myvec{c_0\\c_1\\c_2\\\vdots\\c_n},

       \end{align}}\\

          \vec{F}=\myvec{b-a\\\frac{b^2-a^2}{2}\\\frac{b^3-a^3}{3}\\\vdots\\\frac{b^{n+1}-a^{n+1}}{n+1}}
       \end{align}}\\

      $D(p)=c_1+2c_2x+3c_3x^2+\hdots+nc_nx^{n-1}$ &

       \end{align}}\\
