   \begin{align}

   \end{align}

   \begin{align}
  \myvec{2&-1&\frac{4}{3}&-1&0\\1&0&\frac{2}{3}&0&-1\\9&-3&6&-3&-3}\vec{x}=0
   \end{align}

   \begin{align}

\begin{align}
